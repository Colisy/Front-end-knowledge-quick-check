  2. 组件通信

    emit/props的语法糖v-model

    <input :value="msg” @input="msg=$event.target.value" />

    子组件

    //改变值是通过改变父组件的值，然后传过来展示
currentVal: {
  //有设置就要有get
  //父组件通过props传过来的
   get() {
     return this.value;
   },
   //设置的时候更改父组件数据
   set(val) {
     this.$emit("input", val);
   }
 }
